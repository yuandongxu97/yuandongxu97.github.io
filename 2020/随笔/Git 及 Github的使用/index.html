push时不替换换行符

本地创建推送流程

mkdir study
cd study
git init
touch README.md
git add .
git commit
git remote add origin https://github.com/yuandongxu97/study
git push -u origin master

Git常用命令

config 配置相关

git config --list
git config --global --list

init 创建/初始化一个仓库

clone 克隆仓库到本地目录

add 将文件添加到暂存区

git add file    # 添加file
git add .        # 添加当前目录所有已修改

commit 将暂存区提交到存储库

git commit -a            # git add + git commit    
git commit -m “备注”      # 为提交节点添加备注

diff 当前文件与暂存区域差异

git diff –staged        # 暂存区和存储库差异
git diff [first branch] [second branch]        # 分支差异

reset 撤销操作

git reset [file]        #

status 仓库状态

log 提交日志

git log
git log [file]

show 显示对象

branch 分支的显示、创建与删除

git branch        # 显示本地分支列表
git branch <branch_name>        # 创建新分支
git branch -d <branch_name>        # 删除分支

checkout 分支切换

git checkout <branch_name>        # 切换至
git checkout -b <branch_name>    # 创建并切换至

merge 合并

remote 链接本地与远程库

git remote add [variable name] [Remote Server Link]

push 推送本地至远程仓库

git push [variable name] master        # 推送主分支
git push [variable name] [branch]    # 推送指定分支
git push –all [variable name]        # 推送所有分支
    # 删除远程存储库上的分支

pull 拉取并集合

git pull [Repository Link]
